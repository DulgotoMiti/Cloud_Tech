<!DOCTYPE html>
<html>
     <head>
		<script>


			function loadData()
			{
				id = document.getElementById("userId").value;
				fetch('https://reqres.in/api/users/'+id)
					.then((response) => response.json())
					.then((data) => document.getElementById("res").innerText="Потребител: "+data.data.first_name);
			}


		</script>
     </head>


     <body>
         	<h1>Прегледай потребителите</h1>
		 <hr>
		  
		Идентификатор на потребител: <input type="text" id="userId">
		<button  onclick="loadData()">Зареди</button>
		
		<div id="res"></div>
     </body>
</html>
